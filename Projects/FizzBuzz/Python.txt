for n in range(10**3):
output = ""
                    
# check it against diffrent values
if n % 3 == 0:
    output += "Fizz"
if n % 5 == 0:
    output += "Buzz"
if n % 8 == 0:
    output += "Tozz"
                    
# if none of the above apply
if output == "":
    output = str(n)
                    
print(output)
